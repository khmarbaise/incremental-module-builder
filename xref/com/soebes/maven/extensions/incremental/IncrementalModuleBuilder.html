50  /**
51   * Incremental Module Builder behaviour.
52   * 
53   * @author Karl Heinz Marbaise <khmarbaise@apache.org>
54   * 
55   */
56  @Singleton
57  @Named("incremental")
58  public class IncrementalModuleBuilder implements Builder {
59      private final Logger LOGGER = LoggerFactory.getLogger(getClass());
60  
61      private final LifecycleModuleBuilder lifecycleModuleBuilder;
62  
63      @Inject
64      private ScmManager scmManager;
65  
66      @Inject
67      public IncrementalModuleBuilder(LifecycleModuleBuilder lifecycleModuleBuilder) {
68  	LOGGER.info(" ------------------------------------");
69  	LOGGER.info(" Maven Incremental Module Builder");
70  	LOGGER.info(" Version: {}", IncrementalModuleBuilderVersion.getVersion());
71  	LOGGER.debug("     SHA: {}", IncrementalModuleBuilderVersion.getRevision());
72  	LOGGER.info(" ------------------------------------");
73  	this.lifecycleModuleBuilder = lifecycleModuleBuilder;
74      }
75  
76      private boolean havingScmDeveloperConnection(MavenSession session) {
77  	if (session.getTopLevelProject().getScm() == null) {
78  	    LOGGER.error("The incremental module builder needs a correct scm configuration.");
79  	    return false;
80  	}
81  
82  	if (StringUtils.isEmpty(session.getTopLevelProject().getScm().getDeveloperConnection())) {
83  	    LOGGER.error("The incremental module builder needs the scm developerConnection to work properly.");
84  	    return false;
85  	}
86  
87  	return true;
88      }
89  
90      @Override
91      public void build(final MavenSession session, final ReactorContext reactorContext, ProjectBuildList projectBuilds,
92  	    final List<TaskSegment> taskSegments, ReactorBuildStatus reactorBuildStatus)
93  	    throws ExecutionException, InterruptedException {
94  
95  	// Think about this?
96  	if (!session.getCurrentProject().isExecutionRoot()) {
97  	    LOGGER.info("Not executing in root.");
98  	}
99  
100 	Path projectRootpath = session.getTopLevelProject().getBasedir().toPath();
101 
102 	if (!havingScmDeveloperConnection(session)) {
103 	    return;
104 	}
105 
106 	// TODO: Make more separation of concerns..(Extract the SCM Code from
107 	// here?
108 	ScmRepository repository = null;
109 	try {
110 	    // Assumption: top level project contains the SCM entry.
111 	    repository = scmManager.makeScmRepository(session.getTopLevelProject().getScm().getDeveloperConnection());
112 	} catch (ScmRepositoryException | NoSuchScmProviderException e) {
113 	    LOGGER.error("Failure during makeScmRepository", e);
114 	    return;
115 	}
116 
117 	StatusScmResult result = null;
118 	try {
119 	    result = scmManager.status(repository, new ScmFileSet(session.getTopLevelProject().getBasedir()));
120 	} catch (ScmException e) {
121 	    LOGGER.error("Failure during status", e);
122 	    return;
123 	}
124 
125 	List<ScmFile> changedFiles = result.getChangedFiles();
126 	if (changedFiles.isEmpty()) {
127 	    LOGGER.info(" Nothing has been changed.");
128 	} else {
129 
130 	    for (ScmFile scmFile : changedFiles) {
131 		LOGGER.info(" Changed file: " + scmFile.getPath() + " " + scmFile.getStatus());
132 	    }
133 
134 	    ModuleCalculator mc = new ModuleCalculator(session.getProjectDependencyGraph().getSortedProjects(),
135 		    changedFiles);
136 	    List<MavenProject> calculateChangedModules = mc.calculateChangedModules(projectRootpath);
137 
138 	    for (MavenProject mavenProject : calculateChangedModules) {
139 		LOGGER.info("Changed Project: " + mavenProject.getId());
140 	    }
141 
142 	    IncrementalModuleBuilderImpl incrementalModuleBuilderImpl = new IncrementalModuleBuilderImpl(
143 		    calculateChangedModules, lifecycleModuleBuilder, session, reactorContext, taskSegments);
144 
145 	    // Really build only changed modules.
146 	    incrementalModuleBuilderImpl.build();
147 	}
148     }
